    // Compact de-DE formatters for chart Y-axis ticks.
    // Inputs are ALREADY-converted: EUR for formatEURShort, integers for formatIntShort.
    // Cents-to-EUR conversion is the caller's responsibility (see CalendarRevenueCard
    // chartData map + the `*_eur` fields in CohortRevenueCard/CohortAvgLtvCard).
    export function formatEURShort(eur: number): string {
      return new Intl.NumberFormat('de-DE', {
        notation: 'compact',
        style: 'currency',
        currency: 'EUR',
        maximumFractionDigits: 1
      }).format(eur);
    }

    export function formatIntShort(n: number): string {
      return new Intl.NumberFormat('de-DE', {
        notation: 'compact',
        maximumFractionDigits: 1
      }).format(n);
    }

    Replacement:
    ```svelte
    const chartData = $derived.by(() => {
      const filtered = getFiltered();
      const grain = getFilters().grain as 'day' | 'week' | 'month';
      const nested = aggregateByBucketAndVisitSeq(filtered, grain);
      // Convert cents → EUR integer per series column so the Y-axis renders euros.
      // shapeForChart emits one row per bucket with a numeric value for each visit_seq
      // key (VISIT_KEYS) plus 'cash'. We must walk every such key on every row.
      const SERIES_KEYS = [...VISIT_KEYS, 'cash'] as const;
      return shapeForChart(nested, 'revenue_cents').map((r) => {
        const row: Record<string, string | number> = {
          ...r,
          bucket: formatBucketLabel(r.bucket as string, grain)
        };
        for (const k of SERIES_KEYS) {
          const v = r[k];
          row[k] = typeof v === 'number' ? Math.round(v / 100) : 0;
        }
        return row;
      });
    });
    ```

    Reference pattern: `CohortRevenueCard.svelte:28` already does `Math.round(total_revenue_cents / 100)` — same approach, adapted to stacked shape.
