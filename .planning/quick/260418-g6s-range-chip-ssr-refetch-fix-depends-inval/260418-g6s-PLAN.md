   ```ts

     try {

     } finally {

     }
   ```

   Rationale: current `setTimeout(() => isUpdating = false, 300)` clears the spinner on a fixed 300ms timer regardless of whether the inner work completed. After the fix, cache-miss clicks need to keep the spinner up for the full SSR round-trip (300-1000ms). Removing `setTimeout` and using `try/finally` around `await fn()` makes the spinner duration track the actual work.

3. Make `handleRangeChange` async and add `await invalidate('app:dashboard')` in the cache-miss branch:
   ```ts
   async function handleRangeChange(rangeValue: string) {
     await withUpdate(async () => {
       let window: RangeWindow;
       if (rangeValue === 'custom') {
         const url = new URL(globalThis.window.location.href);
         const from = url.searchParams.get('from')!;
         const to = url.searchParams.get('to')!;
         window = customToRange({ from, to });
         setRangeId('custom', { from, to });
       } else {
         window = chipToRange(rangeValue as Range);
         setRangeId(rangeValue as FiltersState['range']);
       }

       const allFrom = window.priorFrom && window.priorFrom < window.from
         ? window.priorFrom : window.from;

       if (cacheCovers(allFrom, window.to)) {
         setRange(window);
         return;
       }

       // Cache miss — force SSR refetch via the registered invalidation key.
       setRange(window);
       await invalidate('app:dashboard');
     });
   }
   ```

4. Do NOT touch `handleSalesType` or `handleCashFilter` — they stay synchronous (no SSR needed; they're pure client-side store updates via replaceState + set*Filter). `withUpdate` accepts `void | Promise<void>`, so their existing `() => { ... }` shape still compiles.
