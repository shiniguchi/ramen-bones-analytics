       ```ts
       import { z } from 'zod';

       export const RANGE_VALUES = ['today', '7d', '30d', '90d', 'all', 'custom'] as const;
       export const GRAIN_VALUES = ['day', 'week', 'month'] as const;
       export const SALES_TYPE_VALUES = ['INHOUSE', 'TAKEAWAY'] as const;

       export const FILTER_DEFAULTS = Object.freeze({
         range: '7d' as const,
         grain: 'week' as const
       });

       const isoDate = z.string().regex(/^\d{4}-\d{2}-\d{2}$/);
       const csvArray = (allowed?: readonly string[]) =>
         z.string()
           .transform(s => s.split(',').map(x => x.trim()).filter(Boolean))
           .pipe(
             z.array(allowed ? z.enum(allowed as [string, ...string[]]) : z.string().min(1))
           )
           .catch(() => undefined as unknown as string[]);

       export const filtersSchema = z.object({
         range: z.enum(RANGE_VALUES).catch(FILTER_DEFAULTS.range),
         grain: z.enum(GRAIN_VALUES).catch(FILTER_DEFAULTS.grain),
         sales_type: csvArray(SALES_TYPE_VALUES).optional(),
         payment_method: csvArray().optional(),
         from: isoDate.optional().catch(() => undefined),
         to: isoDate.optional().catch(() => undefined)
       });

       export type FiltersState = z.infer<typeof filtersSchema>;

       export function parseFilters(url: URL): FiltersState {
         const raw: Record<string, string> = {};
         for (const [k, v] of url.searchParams) {
           if (v !== '') raw[k] = v;
         }
         return filtersSchema.parse(raw);
       }
       ```
       Note: `csvArray` uses `.catch(() => undefined)` so a single invalid enum value collapses the whole array to undefined rather than partially matching (D-17). Acceptable for v1; Phase 10 may refine.

       ```ts
       export interface CustomRangeInput { from: string; to: string; }

       export function customToRange(input: CustomRangeInput): RangeWindow {
         // Swap if inverted (D-17 tolerance).
         const [lo, hi] = input.from <= input.to
           ? [input.from, input.to]
           : [input.to, input.from];
         // Days length of current window (inclusive).
         const loDate = new Date(lo + 'T00:00:00Z');
         const hiDate = new Date(hi + 'T00:00:00Z');
         const days = Math.round((hiDate.getTime() - loDate.getTime()) / 86400000) + 1;
         const priorHi = new Date(loDate.getTime() - 86400000);
         const priorLo = new Date(priorHi.getTime() - (days - 1) * 86400000);
         const isoUtc = (d: Date) => d.toISOString().slice(0, 10);
         return {
           from: lo,
           to: hi,
           priorFrom: isoUtc(priorLo),
           priorTo: isoUtc(priorHi)
         };
       }
       ```
       Rationale: for literal YYYY-MM-DD input the UTC-midnight trick keeps ISO strings stable without re-entering `toZonedTime` — the user picked Berlin dates already, we don't re-shift.
