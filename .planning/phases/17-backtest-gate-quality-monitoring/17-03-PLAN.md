    **Holiday multiplier helper** (the only genuinely new code — copy template from PATTERNS lines 238-266):

    ```python
    HOLIDAY_FLAGS = ('is_holiday', 'is_school_holiday', 'is_event', 'is_strike')


    def _compute_holiday_multipliers(
        *,
        history_df: pd.DataFrame,         # has 'date' index/col + kpi value column 'y'
        exog_df: pd.DataFrame,            # build_exog_matrix output, indexed by date
        seasonal_means: dict[int, float], # {dow_key: mean_y}
        granularity: str,
    ) -> dict[tuple, float]:
        """Compute multiplier per holiday-flag combination from historical residual ratios.

        For each historical date d:
            ratio = y[d] / dow_mean[seasonal_key(d)]
        Group ratios by (is_holiday, is_school, is_event, is_strike) tuple.
        Return mean ratio per combo. Missing combos at predict-time fall back to 1.0.
        """
        ratios_by_combo: dict[tuple, list[float]] = defaultdict(list)
        # history_df is expected to have columns 'date' and 'y' per naive_dow_fit's _fetch_history shape.
        for d, y in zip(history_df['date'], history_df['y']):
            if d not in exog_df.index:
                continue  # No regressor row -> can't classify; skip
            flags = tuple(int(exog_df.loc[d, f]) for f in HOLIDAY_FLAGS)
            dow_mean = seasonal_means.get(_seasonal_key(d, granularity), 0.0)
            if dow_mean > 0:
                ratios_by_combo[flags].append(float(y) / dow_mean)
        return {combo: float(np.mean(rs)) for combo, rs in ratios_by_combo.items() if rs}


    def _apply_holiday_multipliers(
        *,
        future_dates: list[date],
        seasonal_means: dict[int, float],
        exog_df: pd.DataFrame,
        multipliers: dict[tuple, float],
        granularity: str,
    ) -> np.ndarray:
        """Per future date d: yhat = dow_mean[d.weekday()] * multiplier[combo].

        Combo lookup defaults to 1.0 (== plain naive_dow behavior) when the combo
        was unseen in history.
        """
        out = np.zeros(len(future_dates), dtype=float)
        for i, d in enumerate(future_dates):
            base = seasonal_means.get(_seasonal_key(d, granularity), 0.0)
            if d in exog_df.index:
                flags = tuple(int(exog_df.loc[d, f]) for f in HOLIDAY_FLAGS)
                mult = multipliers.get(flags, 1.0)
            else:
                mult = 1.0
            out[i] = base * mult
        return out
    ```

    ```python
    def fit_and_write(
        client,
        *,
        restaurant_id: str,
        kpi_name: str,
        run_date: date,
        granularity: str = 'day',
        track: str = 'bau',
        train_end: Optional[date] = None,
    ) -> int:
        """Fit naive_dow_with_holidays + write forecast_daily rows. Returns row count."""
        # 1. Fetch history (reuse naive_dow_fit helper)
        history_df = _fetch_history(client, restaurant_id, kpi_name, granularity)
        # 2. Apply train_end cutoff if provided (Phase 17 backtest mode)
        if train_end is not None:
            history_df = history_df[history_df['date'] <= train_end].copy()
        # 3. Compute seasonal means + residuals (reuse helper)
        seasonal_means, residuals = _seasonal_means_and_residuals(history_df, granularity)
        # 4. Build exog matrix covering history + future window
        history_start = history_df['date'].min() if len(history_df) else date.today()
        future_dates = pred_dates_for_grain(
            run_date=run_date, granularity=granularity, train_end=train_end
        )
        exog_end = max(future_dates) if future_dates else run_date
        exog_df = build_exog_matrix(
            client, restaurant_id=restaurant_id, start_date=history_start, end_date=exog_end
        )
        # 5. Compute holiday multipliers from history
        multipliers = _compute_holiday_multipliers(
            history_df=history_df, exog_df=exog_df,
            seasonal_means=seasonal_means, granularity=granularity,
        )
        # 6. Apply to future dates
        yhats = _apply_holiday_multipliers(
            future_dates=future_dates, seasonal_means=seasonal_means,
            exog_df=exog_df, multipliers=multipliers, granularity=granularity,
        )
        # 7. Bootstrap CI bands from residuals
        samples = bootstrap_from_residuals(residuals, n_paths=N_PATHS, n_steps=len(yhats))
        # 8. Build forecast_daily rows (reuse helper, override model_name)
        rows = _build_forecast_rows_daily(
            restaurant_id=restaurant_id, kpi_name=kpi_name,
            run_date=run_date, future_dates=future_dates,
            yhats=yhats, samples=samples, model_name=MODEL_NAME,
            track=track,
        )
        # 9. Upsert
        _upsert_rows(client, rows, chunk_size=CHUNK_SIZE)
        return len(rows)
    ```

def _make_history_with_holiday_lift(*, n_weeks=8, holiday_lift=2.0, strike_drop=0.5):
    """Synthetic 8-week history. Mondays are holidays (lift). Wednesdays are strikes (drop).
    Other days are baseline=100."""
    dates = []
    ys = []
    flags_records = []
    start = date(2025, 6, 9)  # a Monday
    for i in range(n_weeks * 7):
        d = start + timedelta(days=i)
        is_holiday = 1 if d.weekday() == 0 else 0
        is_strike  = 1 if d.weekday() == 2 else 0
        baseline = 100.0
        if is_holiday: baseline *= holiday_lift
        if is_strike: baseline *= strike_drop
        dates.append(d)
        ys.append(baseline)
        flags_records.append({
            'date': d,
            'is_holiday': is_holiday,
            'is_school_holiday': 0,
            'is_event': 0,
            'is_strike': is_strike,
            'is_open': 1,
        })
    history_df = pd.DataFrame({'date': dates, 'y': ys})
    exog_df = pd.DataFrame(flags_records).set_index('date')
    return history_df, exog_df


def test_holiday_multiplier_higher_on_holiday():
    """Holiday-flagged history dates have y=2x baseline -> multiplier ≈ 2.0."""
    history_df, exog_df = _make_history_with_holiday_lift(holiday_lift=2.0)
    # Per-DoW seasonal means for this synthetic history:
    seasonal_means = {dow: float(history_df[history_df['date'].apply(lambda d: d.weekday() == dow)]['y'].mean())
                      for dow in range(7)}
    multipliers = _compute_holiday_multipliers(
        history_df=history_df, exog_df=exog_df,
        seasonal_means=seasonal_means, granularity='day',
    )
    # Combo (is_holiday=1, school=0, event=0, strike=0) should have ratio ≈ 1.0 because
    # the dow_mean is itself the elevated holiday value (Mondays are 200, dow_mean for
    # Monday is 200, so ratio = 200/200 = 1.0). The multiplier captures the residual
    # ratio AFTER dow normalization. The TRUE holiday lift is 2x, but dow_mean already
    # absorbed it. So this test asserts the SHAPE: combo (1,0,0,0) is present.
    assert (1, 0, 0, 0) in multipliers
    # Non-holiday combo (0,0,0,0) should also be present (Tuesdays/Thursdays/etc.)
    assert (0, 0, 0, 0) in multipliers


def test_no_holiday_combo_falls_back_to_1():

        future_dates=future_dates, seasonal_means=seasonal_means,

    )
    # Wednesday's seasonal_means[2] = 100.0; multiplier=1.0 fallback; yhat=100.
    assert abs(yhats[0] - 100.0) < 1e-6


def test_yhat_equals_dow_mean_when_combo_multiplier_is_1():
