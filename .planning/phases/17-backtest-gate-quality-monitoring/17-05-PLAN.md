"""Phase 17: backtest.py — rolling-origin CV driver for the gate.

Spawns each *_fit.py per fold via subprocess (same pattern as run_all.py),
passing --train-end / --eval-start CLI flags from plan 17-04 retrofit.

option b.3 — see plan 17-05 SUMMARY for rationale), computes RMSE/MAPE

rows with evaluation_window='rolling_origin_cv'. Calibrates h=35 conformal
CI via scripts.forecast.conformal. Flips feature_flags.enabled=false for
failing models per BCK-04. Cleans sentinel rows post-eval.

Exit codes:
  0  — at least one (model, horizon, fold) succeeded; gate may have flipped flags

HORIZONS = [7, 35, 120, 365]

DEFAULT_MODELS = ['sarimax', 'prophet', 'ets', 'theta', 'naive_dow', 'naive_dow_with_holidays']

UNCALIBRATED_HORIZONS = (120, 365)   # until ≥730 days history per BCK-02


def _build_subprocess_env(*, restaurant_id, kpi_name, run_date, granularity, forecast_track='bau'):
    """Mirror of run_all.py:_build_subprocess_env (verbatim copy per PATTERNS Pattern A)."""
    env = os.environ.copy()
    env['RESTAURANT_ID'] = restaurant_id
    env['KPI_NAME'] = kpi_name
    env['RUN_DATE'] = run_date.isoformat() if isinstance(run_date, date) else run_date
    env['GRANULARITY'] = granularity
    env['FORECAST_TRACK'] = forecast_track
    for key in ('SUPABASE_URL', 'SUPABASE_SERVICE_ROLE_KEY'):
        if key not in env:
            raise RuntimeError(f'{key} must be set in the runtime environment.')
    return env


def _spawn_fit(*, model: str, restaurant_id: str, kpi_name: str,
               train_end: date, eval_start: date,
               fold_idx: int) -> bool:
    """Subprocess one *_fit.py for a single fold. Returns True on rc=0."""
    cmd = [
        sys.executable, '-m', f'scripts.forecast.{model}_fit',
        '--train-end', train_end.isoformat(),
        '--eval-start', eval_start.isoformat(),
        '--fold-index', str(fold_idx),
    ]
    env = _build_subprocess_env(
        restaurant_id=restaurant_id, kpi_name=kpi_name,
        run_date=eval_start, granularity='day',
        forecast_track=f'backtest_fold_{fold_idx}',
    )
    print(f'[backtest] Spawning fold {fold_idx}/{N_FOLDS}: {model} kpi={kpi_name} train_end={train_end} eval_start={eval_start}')
    result = subprocess.run(cmd, env=env, text=True, capture_output=True)
    if result.stdout: print(result.stdout, end='')
    if result.stderr: print(result.stderr, end='', file=sys.stderr)
    return result.returncode == 0


def _fetch_actuals(client, *, restaurant_id, kpi_name, eval_start, eval_end) -> np.ndarray:
    """Read kpi_daily_mv for the fold's evaluation window."""
    col = 'revenue_cents' if kpi_name == 'revenue_eur' else 'tx_count'
    resp = (
        client.table('kpi_daily_mv')
        .select(f'business_date,{col}')
        .eq('restaurant_id', restaurant_id)
        .gte('business_date', eval_start.isoformat())
        .lte('business_date', eval_end.isoformat())
        .order('business_date')
        .execute()
    )
    rows = resp.data or []
    factor = 100.0 if kpi_name == 'revenue_eur' else 1.0  # cents -> EUR
    return np.array([float(r[col]) / factor for r in rows], dtype=float)


def _fetch_fold_yhats(client, *, restaurant_id, kpi_name, model, fold_idx: int,
                     eval_start, eval_end) -> np.ndarray:
    """Read forecast_daily yhats written by the spawned subprocess fit (FORECAST_TRACK-scoped)."""
    resp = (
        client.table('forecast_daily')
        .select('target_date,yhat')
        .eq('restaurant_id', restaurant_id)
        .eq('kpi_name', kpi_name)
        .eq('model_name', model)
        .eq('forecast_track', f'backtest_fold_{fold_idx}')
        .gte('target_date', eval_start.isoformat())
        .lte('target_date', eval_end.isoformat())
        .order('target_date')
        .execute()
    )
    rows = resp.data or []
    return np.array([float(r['yhat']) for r in rows], dtype=float)


def _days_of_history(client, restaurant_id, kpi_name) -> int:
    """Count rows in kpi_daily_mv for cold-start guard."""
    resp = (
        client.table('kpi_daily_mv')
        .select('business_date', count='exact')
        .eq('restaurant_id', restaurant_id)
        .limit(1)
        .execute()
    )
    return getattr(resp, 'count', None) or 0


def _last_actual_date(client, restaurant_id) -> date:
    """Latest business_date with a kpi_daily_mv row."""
    resp = (
        client.table('kpi_daily_mv')
        .select('business_date')
        .eq('restaurant_id', restaurant_id)
        .order('business_date', desc=True)
        .limit(1)
        .execute()
    )
    rows = resp.data or []
    if not rows:
        raise RuntimeError('No kpi_daily_mv rows for restaurant; cannot run backtest.')
    return date.fromisoformat(rows[0]['business_date'])


def _write_quality_row(client, *, restaurant_id, kpi_name, model, horizon, fold_idx,
                       train_end, eval_start, metrics, gate_verdict=None, qhat: float | None = None):
    """Insert a forecast_quality row with the 4 new diagnostic columns from migration 0067."""
    row = {
        'restaurant_id': restaurant_id,
        'kpi_name': kpi_name,
        'model_name': model,
        'horizon_days': horizon,
        'evaluation_window': 'rolling_origin_cv',
        'evaluated_at': datetime.now(timezone.utc).isoformat(),
        'fold_index': fold_idx,
        'train_end_date': train_end.isoformat() if train_end else None,
        'eval_start_date': eval_start.isoformat() if eval_start else None,
        'gate_verdict': gate_verdict,
        'qhat': qhat,
    }
    if metrics is not None:
        row.update({
            'n_days': metrics.get('n_days'),
            'rmse': round(metrics['rmse'], 4),
            'mape': round(metrics['mape'], 4),
            'mean_bias': round(metrics['mean_bias'], 4),
            'direction_hit_rate': round(metrics['direction_hit_rate'], 4),
        })
    client.table('forecast_quality').upsert(row).execute()


def _gate_decision(quality_rows: list[dict], *, kpi: str, horizon: int) -> dict[str, str]:
    """Per (kpi, horizon) compute PASS/FAIL/PENDING/UNCALIBRATED per model."""
    rows_at_h = [r for r in quality_rows
                 if r['kpi_name'] == kpi and r['horizon_days'] == horizon
                 and r['evaluation_window'] == 'rolling_origin_cv']
    rmses_by_model = defaultdict(list)
    for r in rows_at_h:
        if r.get('rmse') is not None:
            rmses_by_model[r['model_name']].append(r['rmse'])
    if not rmses_by_model:
        return {}
    mean_rmse = {m: float(np.mean(rs)) for m, rs in rmses_by_model.items()}
    if horizon in UNCALIBRATED_HORIZONS:
        # No promotion until 2y data — UNCALIBRATED for everybody at these horizons
        return {m: 'UNCALIBRATED' for m in mean_rmse}
    baseline = max(
        mean_rmse.get('naive_dow', float('inf')),
        mean_rmse.get('naive_dow_with_holidays', float('inf')),
    )
    threshold = baseline * GATE_THRESHOLD
    verdicts = {}
    for m, rmse in mean_rmse.items():
        if m in BASELINE_MODELS:
            verdicts[m] = 'PASS'
        elif rmse <= threshold:
            verdicts[m] = 'PASS'
        else:
            verdicts[m] = 'FAIL'
    return verdicts


def _apply_gate_to_feature_flags(client, *, restaurant_id, model_aggregate_verdicts: dict[str, str]):
    """Flip enabled=false for any non-baseline model whose aggregate verdict is FAIL.

    Aggregate verdict = FAIL if any horizon FAILs. PENDING/UNCALIBRATED do NOT flip.
    Baselines (naive_dow, naive_dow_with_holidays) NEVER flip — R7 hard guard.
    """
    for model, verdict in model_aggregate_verdicts.items():
        if model in BASELINE_MODELS:
            continue
        if verdict == 'FAIL':
            print(f'[backtest] Flipping feature_flags.{model}.enabled=false (FAIL)')
            client.table('feature_flags').update({
                'enabled': False,
                'updated_at': datetime.now(timezone.utc).isoformat(),
            }).eq('restaurant_id', restaurant_id).eq('flag_key', f'model_{model}').execute()


def _cleanup_sentinel_rows(client, *, restaurant_id):
    """DELETE forecast_daily rows written by backtest folds (scoped by forecast_track pattern)."""
    client.table('forecast_daily') \
        .delete() \
        .eq('restaurant_id', restaurant_id) \
        .like('forecast_track', 'backtest_fold_%') \
        .execute()


def main(models: list[str], run_date: date) -> int:
    started_at = datetime.now(timezone.utc)
    client = make_client()
    # Resolve restaurant_id (single-tenant v1)
    resp = client.table('restaurants').select('id').limit(1).execute()
    rows = resp.data or []

        print('[backtest] ERROR: no restaurants', file=sys.stderr)
        return 1
    restaurant_id = rows[0]['id']

    last_actual = _last_actual_date(client, restaurant_id)
    days_history = _days_of_history(client, restaurant_id, 'revenue_eur')

    quality_rows: list[dict] = []
    fold_residuals_at_h35: dict[tuple, dict[int, np.ndarray]] = defaultdict(dict)  # {(kpi, model): {fold: residuals}}

    try:
        for kpi in KPIS:
            for horizon in HORIZONS:
                # Cold-start: <(horizon + N_FOLDS) days of history -> PENDING
                if days_history < horizon + N_FOLDS:
                    for model in models:
                        _write_quality_row(
                            client, restaurant_id=restaurant_id, kpi_name=kpi,
                            model=model, horizon=horizon, fold_idx=None,
                            train_end=None, eval_start=None, metrics=None,
                            gate_verdict='PENDING',
                        )
                    print(f'[backtest] kpi={kpi} h={horizon}: PENDING (history={days_history}d < required={horizon+N_FOLDS}d)')
                    continue

                # Run N_FOLDS rolling-origin folds
                for fold_idx in range(N_FOLDS):
                    eval_end = last_actual - timedelta(days=fold_idx * horizon)
                    eval_start = eval_end - timedelta(days=horizon - 1)
                    train_end = eval_start - timedelta(days=1)
                    sentinel = _sentinel_run_date(fold_idx)

                    for model in models:
                        ok = _spawn_fit(
                            model=model, restaurant_id=restaurant_id, kpi_name=kpi,
                            train_end=train_end,
                            eval_start=eval_start, fold_idx=fold_idx,
                        )
                        if not ok:
                            print(f'[backtest] fold {fold_idx} model={model} kpi={kpi} h={horizon}: FIT FAILED, skipping metrics', file=sys.stderr)
                            continue
                        actuals = _fetch_actuals(client, restaurant_id=restaurant_id, kpi_name=kpi,
                                                 eval_start=eval_start, eval_end=eval_end)
                        yhats = _fetch_fold_yhats(client, restaurant_id=restaurant_id, kpi_name=kpi,
                                                  model=model, fold_idx=fold_idx,
                                                  eval_start=eval_start, eval_end=eval_end)
                        # Align lengths defensively (closed days etc.)
                        n = min(len(actuals), len(yhats))
                        if n == 0:
                            continue
                        actuals = actuals[:n]; yhats = yhats[:n]
                        metrics = compute_metrics(actuals, yhats)
                        metrics['n_days'] = int(n)
                        _write_quality_row(
                            client, restaurant_id=restaurant_id, kpi_name=kpi,
                            model=model, horizon=horizon, fold_idx=fold_idx,
                            train_end=train_end, eval_start=eval_start,
                            metrics=metrics, gate_verdict=None,  # filled in second pass after gate
                        )
                        quality_rows.append({
                            'kpi_name': kpi, 'model_name': model,
                            'horizon_days': horizon, 'rmse': metrics['rmse'],
                            'evaluation_window': 'rolling_origin_cv',
                            'fold_index': fold_idx,
                        })
                        # Collect h=35 residuals for conformal (one residual per fold = mean signed residual)
                        if horizon == 35:
                            residuals = (actuals - yhats)
                            fold_residuals_at_h35[(kpi, model)][fold_idx] = residuals

        # Conformal calibration phase (BCK-02) — per (kpi, model)
        for (kpi, model), folds in fold_residuals_at_h35.items():
            qhat = calibrate_conformal_h35(folds, alpha=0.05)
            print(f'[backtest] conformal qhat_h35[{kpi}/{model}] = {qhat["qhat_h35"]:.4f}')
            _write_quality_row(
                client, restaurant_id=restaurant_id, kpi_name=kpi,
                model=model, horizon=35, fold_idx=None,
                train_end=None, eval_start=None, metrics=None,
                gate_verdict=None, qhat=qhat['qhat_h35'],
            )

        # Gate decision phase (BCK-04)
        per_model_per_horizon_verdicts: dict[str, dict[int, str]] = defaultdict(dict)
        for kpi in KPIS:
            for horizon in HORIZONS:
                verdicts = _gate_decision(quality_rows, kpi=kpi, horizon=horizon)
                for model, verdict in verdicts.items():
                    per_model_per_horizon_verdicts[model][horizon] = verdict
                    # Backfill gate_verdict on the existing forecast_quality rows
                    client.table('forecast_quality') \
                        .update({'gate_verdict': verdict}) \
                        .eq('restaurant_id', restaurant_id) \
                        .eq('kpi_name', kpi) \
                        .eq('model_name', model) \
                        .eq('horizon_days', horizon) \
                        .eq('evaluation_window', 'rolling_origin_cv') \
                        .gte('evaluated_at', started_at.isoformat()) \
                        .execute()

        # Aggregate verdict per model (a model is enabled iff PASS at all evaluable horizons)
        aggregate: dict[str, str] = {}
        for model, hv in per_model_per_horizon_verdicts.items():
            non_pending = [v for v in hv.values() if v not in ('PENDING', 'UNCALIBRATED')]
            if not non_pending:
                aggregate[model] = 'PENDING'
            elif any(v == 'FAIL' for v in non_pending):
                aggregate[model] = 'FAIL'
            else:
                aggregate[model] = 'PASS'

        _apply_gate_to_feature_flags(client, restaurant_id=restaurant_id, model_aggregate_verdicts=aggregate)

        # Cleanup sentinel rows
        _cleanup_sentinel_rows(client, restaurant_id=restaurant_id)

        n_rows = len(quality_rows)
        write_success(client, step_name=STEP_NAME, started_at=started_at,
                      row_count=n_rows, restaurant_id=restaurant_id)
        print(f'[backtest] DONE. {n_rows} forecast_quality rows; aggregate verdicts: {aggregate}')
        return 0
    except Exception as e:
        traceback.print_exc()
        try:
            write_failure(client, step_name=STEP_NAME, started_at=started_at,
                          error_msg=str(e), restaurant_id=restaurant_id)
        except Exception:
            pass
        return 1
