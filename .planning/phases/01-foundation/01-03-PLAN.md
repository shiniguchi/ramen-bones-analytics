```sql
-- supabase/migrations/0004_kpi_daily_mv_template.sql

-- 1. Materialized view (placeholder content per D-08a). Owner: postgres.
create materialized view public.kpi_daily_mv as
select
  r.id           as restaurant_id,
  current_date   as business_date,
  0::numeric     as revenue_cents
from public.restaurants r;

-- 2. MANDATORY unique index (D-08). Enables REFRESH ... CONCURRENTLY in Phase 3.
create unique index kpi_daily_mv_pk
  on public.kpi_daily_mv (restaurant_id, business_date);

-- 3. Lock the raw MV. authenticated/anon can never read it directly (D-07).
revoke all on public.kpi_daily_mv from anon, authenticated;

-- 4. Wrapper view — the ONLY tenant-facing read path (D-06).
-- DO NOT add WITH (security_invoker = on) here. Default (off) is load-bearing:
-- the view runs as its owner (postgres), which has SELECT on the REVOKE'd MV.
-- authenticated can SELECT the view; the WHERE clause enforces tenancy via the JWT claim.
-- See Pitfall A for the failure mode if this is changed.
create view public.kpi_daily_v as
select
  restaurant_id,
  business_date,
  revenue_cents
from public.kpi_daily_mv
where restaurant_id::text = (auth.jwt()->>'restaurant_id');

grant select on public.kpi_daily_v to authenticated;
```

Requirements the file MUST satisfy simultaneously (CI guard 3b in Plan 05 checks the first two):
- A `create unique index` statement in the SAME file as the `create materialized view`.
- A `revoke all on public.kpi_daily_mv from anon, authenticated` statement.
- The wrapper view `create view public.kpi_daily_v` with a `where restaurant_id::text = (auth.jwt()->>'restaurant_id')` clause.
- NO `security_invoker = on` anywhere in the file (Pitfall A).
- A `grant select on public.kpi_daily_v to authenticated` statement.
