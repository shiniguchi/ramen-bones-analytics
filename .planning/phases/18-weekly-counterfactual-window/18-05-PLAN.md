```svelte
<!-- Inside <Svg>, after bind:context={chartCtx} on <Chart>: -->
{#each chartData as wk, i}
  {@const x = chartCtx.xScale(wk.iso_week_start)}
  {@const w = chartCtx.xScale.bandwidth()}
  {@const y = chartCtx.yScale(Math.max(wk.point_eur, 0))}
  {@const h = Math.abs(chartCtx.yScale(0) - chartCtx.yScale(wk.point_eur))}
  {@const fill =
    wk.ci_lower_eur > 0 ? 'fill-emerald-500' :
    wk.ci_upper_eur < 0 ? 'fill-rose-500' : 'fill-zinc-400'}
  <rect
    {x} {y} width={w} height={h}
    class={fill}
    rx={2}
    role="button"
    tabindex="0"
    aria-label="Week of {wk.iso_week_start} bar"
    onclick={() => (selectedWeekIndex = i)}
    data-testid="uplift-week-bar"
    data-week-index={i}
  />
{/each}
```
