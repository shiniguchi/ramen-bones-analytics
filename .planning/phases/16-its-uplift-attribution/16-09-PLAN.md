    ```svelte
    <script lang="ts">
      import { Chart, Svg, Spline, Area, Tooltip } from 'layerchart';
      import { scaleTime } from 'd3-scale';
      import { curveMonotoneX } from 'd3-shape';
      import { format, differenceInDays } from 'date-fns';

      type UpliftBlockRow = {
        model_name: string;
        window_kind: 'campaign_window' | 'cumulative_since_launch';
        cumulative_uplift_eur: number;
        ci_lower_eur: number;
        ci_upper_eur: number;
        naive_dow_uplift_eur: number | null;
        n_days: number;
        as_of_date: string;
      };
      type CampaignBlock = {
        campaign_id: string;
        start_date: string;
        end_date: string;
        name: string | null;
        channel: string | null;
        rows: UpliftBlockRow[];
      };

      type DailyPoint = { date: string; cumulative_uplift_eur: number; ci_lower_eur: number; ci_upper_eur: number };

      let data = $state<{ campaigns: CampaignBlock[]; ci_lower_eur: number | null; ci_upper_eur: number | null; naive_dow_uplift_eur: number | null; daily: DailyPoint[] } | null>(null);
      let loading = $state(true);
      let loadError = $state<string | null>(null);

      // Sparkline data — per CONTEXT.md D-11 ("inline 280px-wide cumulative-uplift
      // sparkline (LayerChart Spline + low-opacity Area CI band)" + "shape-of-uplift")
      // we render the FULL per-day trajectory from the API's daily[] field
      // (sourced from campaign_uplift_daily_v, populated by Plan 06 Task 4).
      // Each point carries (date, cum_uplift, ci_lower, ci_upper) so the Spline
      // shows the running cumulative line and the Area shows the per-day CI band.

      $effect(() => {
        (async () => {
          try {
            const res = await fetch('/api/campaign-uplift', { credentials: 'same-origin' });
            if (!res.ok) throw new Error(`HTTP ${res.status}`);
            data = await res.json();
          } catch (e) {
            console.error('[CampaignUpliftCard]', e);
            loadError = e instanceof Error ? e.message : 'fetch failed';
          } finally {
            loading = false;
          }
        })();
      });

      // Derived headline state per D-11 / UPL-06:
      let headline = $derived.by(() => {
        if (!data || data.campaigns.length === 0) return null;
        const c = data.campaigns[0];
        const r = c.rows.find((row) => row.model_name === 'sarimax' && row.window_kind === 'cumulative_since_launch');
        return r ? { campaign: c, row: r } : null;
      });

      // CI overlaps zero rule (UPL-06 honest label)
      let ciOverlapsZero = $derived(
        headline ? headline.row.ci_lower_eur <= 0 && headline.row.ci_upper_eur >= 0 : false
      );

      // Divergence warning (D-09): sign-disagree OR >50% magnitude
      let divergenceWarning = $derived.by(() => {
        if (!headline || headline.row.naive_dow_uplift_eur === null) return false;
        const s = headline.row.cumulative_uplift_eur;
        const n = headline.row.naive_dow_uplift_eur;
        const signDisagree = Math.sign(s) !== Math.sign(n) && (s !== 0 || n !== 0);
        const magnitudeDivergent = Math.abs(s - n) / Math.max(Math.abs(s), 1) > 0.5;
        return signDisagree || magnitudeDivergent;
      });

      // Sparkline source — per-day trajectory consumed directly from API daily[].
      // NEVER synthesize a 2-point line; CONTEXT.md D-11 requires the full
      // shape-of-uplift over every day in the campaign window.
      let sparklineData = $derived.by(() => {
        if (!data || !data.daily || data.daily.length === 0) return [];
        return data.daily.map((d) => ({
          date: new Date(d.date),
          cum_uplift: d.cumulative_uplift_eur,
          ci_lower: d.ci_lower_eur,
          ci_upper: d.ci_upper_eur,
        }));
      });

      function formatEur(v: number): string {
        const sign = v >= 0 ? '+' : '−';
        return `${sign}€${Math.abs(Math.round(v)).toLocaleString('de-DE')}`;
      }
    </script>

    {#if loading}
      <div class="rounded-2xl border bg-white p-4 animate-pulse" data-testid="campaign-uplift-card">
        <div class="h-6 w-48 bg-gray-200 rounded mb-3"></div>
        <div class="h-[100px] w-[280px] bg-gray-100 rounded"></div>
      </div>
    {:else if loadError}
      <div class="rounded-2xl border bg-white p-4 text-sm text-gray-500" data-testid="campaign-uplift-card">
        Could not load uplift.
      </div>
    {:else if !data || data.campaigns.length === 0 || !headline}
      <!-- Empty: hide entirely -->
    {:else}
      <div class="rounded-2xl border bg-white p-4" data-testid="campaign-uplift-card">
        <h2 class="text-base font-semibold mb-1">
          Did the {format(new Date(headline.campaign.start_date), 'MMM d, yyyy')} campaign work?
        </h2>

        {#if ciOverlapsZero}
          <p class="text-lg font-bold" data-testid="hero-ci-overlaps">
            CI overlaps zero — no detectable lift
          </p>
          <p class="text-sm text-gray-500 mt-0.5" data-testid="dim-point-estimate">
            {formatEur(headline.row.cumulative_uplift_eur)} (95% CI {formatEur(headline.row.ci_lower_eur)} … {formatEur(headline.row.ci_upper_eur)})
          </p>
        {:else}
          <p class="text-2xl font-bold" data-testid="hero-uplift">
            Cumulative uplift: {formatEur(headline.row.cumulative_uplift_eur)}
          </p>
          <p class="text-xs text-gray-500 mt-0.5">
            95% CI {formatEur(headline.row.ci_lower_eur)} … {formatEur(headline.row.ci_upper_eur)}
          </p>
        {/if}

        <div class="mt-3" style="width: 280px; height: 100px;">
          <Chart
            data={sparklineData}
            x="date"
            y={['ci_lower', 'ci_upper']}
            xScale={scaleTime()}
            yNice={2}
            padding={{ left: 0, right: 0, top: 4, bottom: 4 }}
            tooltip={{ mode: 'bisect-x' }}
            tooltipContext={{ mode: 'bisect-x', touchEvents: 'auto' }}
          >
            <Svg>
              <Area
                y0="ci_lower"
                y1="ci_upper"
                fill="currentColor"
                fill-opacity={0.06}
                line={false}
                curve={curveMonotoneX}
              />
              <Spline
                y="cum_uplift"
                class="stroke-primary stroke-2"
                curve={curveMonotoneX}
              />
            </Svg>
            <Tooltip.Root contained="window" class="max-w-[80vw] text-xs">
              {#snippet children({ data: pt })}
                <Tooltip.Header value={format(pt.date, 'MMM d')} />
                <Tooltip.List>
                  <Tooltip.Item
                    label={`Day ${differenceInDays(pt.date, new Date(headline.campaign.start_date))}`}
                    value={formatEur(pt.cum_uplift)}
                  />
                  <Tooltip.Item
                    label="95% CI"
                    value={`${formatEur(pt.ci_lower)} … ${formatEur(pt.ci_upper)}`}
                  />
                </Tooltip.List>
              {/snippet}
            </Tooltip.Root>
          </Chart>
        </div>

        {#if divergenceWarning}
          <p class="mt-2 text-xs text-amber-600" data-testid="divergence-warning">
            Naive baseline disagrees — review the methodology.
          </p>
        {/if}

        <p class="mt-2 text-[11px] text-gray-400" data-testid="anticipation-buffer-note">
          Counterfactual fits on data ≥7 days before the campaign start (anticipation buffer).
        </p>
      </div>
    {/if}
    ```
