    ```python
    def get_train_end(client, restaurant_id: str, train_end_offset: int = -7) -> Optional[date]:
        """Returns earliest campaign_calendar.start_date + train_end_offset days.
        Returns None if no campaign exists (no fit possible).
        """
        resp = client.table('campaign_calendar').select('start_date').eq(
            'restaurant_id', restaurant_id
        ).order('start_date', desc=False).limit(1).execute()
        if not resp.data:
            return None
        earliest = date.fromisoformat(resp.data[0]['start_date'])
        return earliest + timedelta(days=train_end_offset)


    def load_history(client, restaurant_id: str, kpi_name: str, train_end: date) -> pd.DataFrame:
        """Source historical data from kpi_daily_with_comparable_v (NEVER kpi_daily_mv)
        capped at train_end (inclusive). Per Guard 9 / D-04.

        Returns DataFrame with columns: business_date, value (numeric).

        # Note: the executor must adapt the column projection — kpi_daily_with_comparable_v
        # exposes revenue_comparable_eur and tx_count. Map kpi_name -> column.
        col_map = {
    ```

    ```python
    def main_cf(*, client, restaurant_id: str, models, run_date: date,
                train_end_offset: int = -7) -> dict:
        """Top-level CF orchestrator. Returns {'attempted': int, 'succeeded': int}."""
        train_end = get_train_end(client, restaurant_id, train_end_offset)
        if train_end is None:
            print('[CF] No campaign_calendar rows for restaurant; skipping CF.', file=sys.stderr)
            return {'attempted': 0, 'succeeded': 0}

        attempted = 0
        succeeded = 0
        models_to_run = models or CF_MODELS
        for model in models_to_run:
            if model not in CF_MODELS:
                continue
            for kpi in CF_KPIS:
                attempted += 1
                started_at = datetime.now(timezone.utc)
                try:
                    n = fit_one_model(
                        client, model=model, restaurant_id=restaurant_id,
                        kpi_name=kpi, run_date=run_date, train_end=train_end,
                    )
                    write_success(
                        client,
                        step_name=f'cf_{model}',
                        started_at=started_at,
                        row_count=n,
                        restaurant_id=restaurant_id,
                        fit_train_end=train_end,
                    )
                    succeeded += 1
                except Exception:
                    err = traceback.format_exc()
                    print(f'[CF {model}/{kpi}] FAIL: {err}', file=sys.stderr)
                    try:
                        write_failure(
                            client,
                            step_name=f'cf_{model}',
                            started_at=started_at,
                            error_msg=err,
                            restaurant_id=restaurant_id,
                            fit_train_end=train_end,
                        )
                    except Exception:
                        pass
                    continue
        return {'attempted': attempted, 'succeeded': succeeded}
    ```
