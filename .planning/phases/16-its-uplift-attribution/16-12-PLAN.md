    Create `tests/forecast/check_cutoff_sensitivity.sh` (executable):

    ```bash
    #!/usr/bin/env bash
    # tests/forecast/check_cutoff_sensitivity.sh
    # Asserts cutoff_sensitivity.md (Phase 16 D-13 / UPL-02) is well-formed:
    #  - 5 model rows present
    #  - 3 cutoff columns (-14, -7, -1) populated for each model
    #  - At least sarimax + prophet have ratio in [0.8, 1.25]

    set -uo pipefail
    REPO="$(cd "$(dirname "$0")/../.." && pwd)"
    LOG="$REPO/tests/forecast/cutoff_sensitivity.md"

    if [ ! -f "$LOG" ]; then
      echo "FAIL: $LOG not found"; exit 1
    fi

    # Check 5 model rows
    rows=$(grep -cE "^\| (sarimax|prophet|ets|theta|naive_dow)" "$LOG")
    if [ "$rows" -ne 5 ]; then
      echo "FAIL: expected 5 model rows, got $rows"; exit 1
    fi

    # Check Sensitivity ratio language present
    if ! grep -q "Sensitivity ratio" "$LOG"; then
      echo "FAIL: 'Sensitivity ratio' methodology section missing"; exit 1
    fi

    # Parse the sarimax + prophet rows; extract the ratio column (8th); assert in [0.8, 1.25].
    # Robustness: a row may say FAIL — that's allowed for ets/theta/naive_dow but
    # NOT for sarimax + prophet (per acceptance band).
    python3 - "$LOG" <<'PYEOF'
import re, sys, pathlib
text = pathlib.Path(sys.argv[1]).read_text()
required = ['sarimax', 'prophet']
ok = True
for model in required:
    # Match the table row starting with | <model>
    m = re.search(rf'^\|\s*{model}\s*\|.*$', text, flags=re.MULTILINE)
    if not m:
        print(f'FAIL: row for {model} missing'); ok = False; continue
    cells = [c.strip() for c in m.group(0).strip('|').split('|')]
    # cells: [model, -14_uplift, -14_ci, -7_uplift, -7_ci, -1_uplift, -1_ci, ratio, verdict]
    if len(cells) < 8:
        print(f'FAIL: {model} row has {len(cells)} cells, need >=8'); ok = False; continue
    ratio_str = cells[7].replace('—', '').replace('FAIL', '')
    try:
        ratio = float(ratio_str)
    except ValueError:
        print(f'FAIL: {model} ratio cell "{cells[7]}" is not numeric'); ok = False; continue
    if not (0.8 <= ratio <= 1.25):
        print(f'WARN: {model} ratio {ratio} outside [0.8, 1.25] band — investigate anticipation effects (RESEARCH §2 Pitfall 2.2). Phase 16 still ships — a sensitivity-log warning is informational, not blocking, per CONTEXT.md.')
        # Not blocking — warning only per CONTEXT.md interpretation.
sys.exit(0 if ok else 1)
PYEOF

    if [ $? -ne 0 ]; then
      echo "FAIL: cutoff_sensitivity.md does not meet sarimax+prophet completeness"
      exit 1
    fi

    echo "PASS: cutoff_sensitivity.md well-formed; sarimax+prophet ratios documented"
    exit 0
    ```

    Make executable: `chmod +x tests/forecast/check_cutoff_sensitivity.sh`.

    Note: per CONTEXT.md, ratio outside [0.8, 1.25] is informational (warn) not blocking — the phase still ships, but the warning surfaces for human review. Adjust this nuance if the user prefers a hard fail.
