    2. View body:
       ```sql
       CREATE OR REPLACE VIEW public.baseline_items_v AS
       WITH first_seen AS (
         SELECT
           restaurant_id,
           item_name,
           MIN(occurred_at::date) AS first_seen_date
         FROM public.stg_orderbird_order_items
         WHERE item_name IS NOT NULL
         GROUP BY restaurant_id, item_name
       ),
       min_campaign AS (
         SELECT
           restaurant_id,
           MIN(start_date) AS earliest_campaign_start
         FROM public.campaign_calendar
         GROUP BY restaurant_id
       )
       SELECT
         fs.restaurant_id,
         fs.item_name,
         fs.first_seen_date
       FROM first_seen fs
       INNER JOIN min_campaign mc ON mc.restaurant_id = fs.restaurant_id
       WHERE fs.first_seen_date <= mc.earliest_campaign_start - INTERVAL '7 days'
         AND fs.restaurant_id = (auth.jwt()->>'restaurant_id')::uuid;

       GRANT SELECT ON public.baseline_items_v TO authenticated;
       ```

    3. `COMMENT ON VIEW public.baseline_items_v IS 'Phase 16 D-02: items first seen >=7d before earliest campaign_start; comparable-revenue baseline.';`
