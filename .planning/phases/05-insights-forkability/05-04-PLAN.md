**Edit 1: `src/routes/+page.server.ts`**

Add ONE more entry to the existing Promise.all (do NOT restructure). Fetch the latest insight:

```typescript
// Inside the existing load function, alongside the other fan-out queries:
const insightQ = supabase
  .from("insights_v")
  .select("id, business_date, headline, body, fallback_used")
  .order("business_date", { ascending: false })
  .limit(1)
  .maybeSingle();

// Add insightQ to the Promise.all array; extract { data: latestInsightRow, error: insightErr } from the result tuple.

// After the Promise.all settles:
const today = new Intl.DateTimeFormat("en-CA", {
  timeZone: session.user.app_metadata?.timezone ?? "Europe/Berlin",
  year: "numeric", month: "2-digit", day: "2-digit",
}).format(new Date());

let latestInsight: {
  headline: string;
  body: string;
  business_date: string;
  fallback_used: boolean;
  is_yesterday: boolean;
} | null = null;

if (latestInsightRow && !insightErr) {
  latestInsight = {
    headline: latestInsightRow.headline,
    body: latestInsightRow.body,
    business_date: latestInsightRow.business_date,
    fallback_used: latestInsightRow.fallback_used,
    is_yesterday: latestInsightRow.business_date !== today,
  };
}
// Silent swallow on error per UI-SPEC "no error state for InsightCard"
if (insightErr) {
  console.error("[+page.server] insights_v fetch failed:", insightErr.message);
}

return {
  // ...existing fields,
  latestInsight,
};
```

Do NOT change any existing fan-out entries. One new query, one new return field.
