# 3. Set topics (one --add-topic per topic; gh supports multiple)
gh repo edit \
  --add-topic analytics \
  --add-topic restaurant-analytics \
  --add-topic sveltekit \
  --add-topic svelte \
  --add-topic supabase \
  --add-topic cloudflare-pages \
  --add-topic forkable \
  --add-topic pos-integration \
  --add-topic cohort-analysis

# 4. Verify
