**hash.ts** — D-07 exact spec:
```ts
import { createHash } from 'node:crypto';
export function hashCard(wl: string | null | undefined, rid: string): string | null {
  if (!wl || wl.trim() === '') return null;
  return createHash('sha256').update(wl + rid).digest('hex');
}
```

**normalize.ts** — four exports:

1. `toStagingRows(rows, restaurantId, sourceFile)`:
   - Group rows by `invoice_number` preserving original file order
   - Assign `row_index` 1..N within each invoice group
   - Map each CSV row to a StagingRow shape (all 29 csv columns go in verbatim as strings; `card_number`/wl_card_number stays because this is the staging mirror — guard 4 applies only when co-referenced with card_hash)

2. `toBerlinUtc(date, time)`:
   ```ts
   import { fromZonedTime } from 'date-fns-tz';
   return fromZonedTime(`${date} ${time}`, 'Europe/Berlin').toISOString();
   ```

3. `normalizePaymentMethod(raw)`:
   - Canonical map: `MASTERCARD|Mastercard → MasterCard`, `VISA → Visa`, `BAR → Bar`, etc.
   - Case-insensitive lookup; fallback to title-case; `Bar` for null/empty when the row is cash.

4. `toTransactions(staging, restaurantId)`:
   - Group by `invoice_number`, take first row of each group
   - Skip groups where `parseFloat(first.invoice_total_eur) < 0` (D-11)
   - Compute fields:
     - `source_tx_id = first.invoice_number` (D-04)
     - `occurred_at = toBerlinUtc(first.csv_date, first.csv_time)`
     - `card_hash = hashCard(first.wl_card_number, restaurantId)` (D-07/D-08)
     - `gross_cents = Math.round(parseFloat(first.invoice_total_eur) * 100)` — use `Math.round` not `Math.floor` (Pitfall 5)
     - `net_cents  = Math.round(parseFloat(first.invoice_total_eur) / (1 + parseFloat(first.tax_rate_pct)/100) * 100)`
     - `tip_cents  = Math.round(parseFloat(first.tip_eur || '0') * 100)` — from FIRST row only (D-12), NEVER sum
     - `payment_method = normalizePaymentMethod(first.payment_method)`
     - `sales_type = first.sales_type`
   - Return TxRow[]

**index.ts** — orchestrator, wired to `npm run ingest`:
```ts
import 'dotenv/config';
import { loadEnv } from './env';
import { downloadCsv } from './download';
import { parseCsv } from './parse';
import { toStagingRows, toTransactions } from './normalize';
import { upsertStaging, upsertTransactions } from './upsert';
import { createClient } from '@supabase/supabase-js';
import type { IngestReport } from './types';

export async function runIngest(opts: { dryRun?: boolean } = {}): Promise<IngestReport> {
  const env = loadEnv();
  const client = createClient(env.SUPABASE_URL, env.SUPABASE_SERVICE_ROLE_KEY, {
    auth: { persistSession: false, autoRefreshToken: false },
  });
  const csvText = await downloadCsv(client, env.ORDERBIRD_CSV_BUCKET, env.ORDERBIRD_CSV_OBJECT);
  const rows = parseCsv(csvText);
  const staging = toStagingRows(rows, env.RESTAURANT_ID, env.ORDERBIRD_CSV_OBJECT);
  const tx = toTransactions(staging, env.RESTAURANT_ID);

  const missingWl = rows.filter(r =>
    (r.wl_card_number ?? '').trim() === '' && (r.payment_method ?? '').toLowerCase() !== 'bar'
  ).length;
  const cashExcluded = rows.filter(r => (r.wl_card_number ?? '').trim() === '').length;

  if (opts.dryRun) {
    return { rows_read: rows.length, invoices_deduped: tx.length, staging_upserted: 0,
             transactions_new: 0, transactions_updated: 0, cash_rows_excluded: cashExcluded,
             missing_worldline_rows: missingWl, errors: 0 };
  }

  const preTxCount = await txCount(client, env.RESTAURANT_ID);
  await upsertStaging(client, staging);
  await upsertTransactions(client, tx);
  const postTxCount = await txCount(client, env.RESTAURANT_ID);

  const report: IngestReport = {
    rows_read: rows.length,
    invoices_deduped: tx.length,
    staging_upserted: staging.length,
    transactions_new: postTxCount - preTxCount,
    transactions_updated: tx.length - (postTxCount - preTxCount),
    cash_rows_excluded: cashExcluded,
    missing_worldline_rows: missingWl,
    errors: 0,
  };
  console.log(JSON.stringify(report));
  return report;
}

// CLI entry
if (import.meta.url === `file://${process.argv[1]}`) {
  runIngest({ dryRun: process.argv.includes('--dry-run') })
    .catch(err => { console.error(err); process.exit(1); });
}
```
