    1. Create `src/lib/dateRange.ts`:
    ```ts
    import { formatISO, subDays, startOfDay } from 'date-fns';
    import { toZonedTime } from 'date-fns-tz';

    export type Range = 'today'|'7d'|'30d'|'90d'|'all';
    export type Grain = 'day'|'week'|'month';
    const TZ = 'Europe/Berlin';

    export function chipToRange(range: Range, now = new Date()) {
      const today = startOfDay(toZonedTime(now, TZ));
      const isoDate = (d: Date) => formatISO(d, { representation: 'date' });
      if (range === 'all') {
        return { from: '1970-01-01', to: isoDate(today), priorFrom: null, priorTo: null };
      }
      const days = range === 'today' ? 1 : range === '7d' ? 7 : range === '30d' ? 30 : 90;
      const from = subDays(today, days - 1);
      const priorTo = subDays(from, 1);
      const priorFrom = subDays(priorTo, days - 1);
      return { from: isoDate(from), to: isoDate(today), priorFrom: isoDate(priorFrom), priorTo: isoDate(priorTo) };
    }
    ```

    ```ts
    import { redirect } from '@sveltejs/kit';
    import type { Actions, PageServerLoad } from './$types';
    import { chipToRange, type Range, type Grain } from '$lib/dateRange';

    export const load: PageServerLoad = async ({ locals, url }) => {
      const range = (url.searchParams.get('range') ?? '7d') as Range;
      const grain = (url.searchParams.get('grain') ?? 'week') as Grain;

      let freshness: string | null = null;
      try {
        const { data } = await locals.supabase
          .from('data_freshness_v')
          .select('last_ingested_at')
          .maybeSingle();
        freshness = data?.last_ingested_at ?? null;
      } catch (err) {
        console.error('[+page.server] data_freshness_v query failed', err);
      }

      return { range, grain, freshness, window: chipToRange(range) };
    };

    export const actions: Actions = {
      logout: async ({ locals }) => {
        await locals.supabase.auth.signOut();
        throw redirect(303, '/login');
      }
    };
    ```
    NOTE: do NOT call `getSession()` directly (Guard 2). `locals.supabase` is already JWT-bound via hooks + layout. The `redirect` import must come from `@sveltejs/kit`; `Actions`/`PageServerLoad` from the generated `./$types`.
