SECURITY DEFINER` but that change is not in the repo.

       import { describe, it, expect, beforeAll } from 'vitest';
       import { createClient } from '@supabase/supabase-js';

       const url = process.env.TEST_SUPABASE_URL!;
       const anon = process.env.TEST_SUPABASE_ANON_KEY!;
       const email = process.env.TEST_USER_EMAIL!;
       const password = process.env.TEST_USER_PASSWORD!;
       const expectedRestaurantId = process.env.TEST_USER_RESTAURANT_ID!;

       function decodeJwt(token: string): Record<string, unknown> {
         const payload = token.split('.')[1];
         const json = Buffer.from(payload, 'base64url').toString('utf8');
         return JSON.parse(json);
       }

       describe('JWT claim regression — Gap B (custom_access_token_hook SECURITY DEFINER)', () => {
         let accessToken: string;

         beforeAll(async () => {
           if (!url || !anon || !email || !password || !expectedRestaurantId) {
             throw new Error('jwt-claim test requires TEST_SUPABASE_URL/ANON_KEY/USER_EMAIL/USER_PASSWORD/USER_RESTAURANT_ID');
           }
           const client = createClient(url, anon);
           const { data, error } = await client.auth.signInWithPassword({ email, password });
           if (error) throw error;
           if (!data.session?.access_token) throw new Error('no access token returned');
           accessToken = data.session.access_token;
         });

         it('access token decodes to a JWT carrying restaurant_id', () => {
           const claims = decodeJwt(accessToken);
           expect(

             'Gap B regression: custom_access_token_hook is not SECURITY DEFINER, so GoTrue read 0 rows from memberships under RLS and minted a JWT without restaurant_id.'

           expect(claims.restaurant_id).toBe(expectedRestaurantId);
         });
       });
